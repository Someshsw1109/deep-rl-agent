model.py
import torch
import torch.nn as nn

class Net(nn.Module):
    def __init__(self, s, a):
        super().__init__()
        self.l1 = nn.Linear(s, 128)
        self.l2 = nn.Linear(128, 128)
        self.l3 = nn.Linear(128, a)

    def forward(self, x):
        x = torch.relu(self.l1(x))
        x = torch.relu(self.l2(x))
        return self.l3(x)
replay_buffer.py
import random
from collections import deque

class Buffer:
    def __init__(self, c):
        self.b = deque(maxlen=c)

    def add(self, x):
        self.b.append(x)

    def sample(self, n):
        return random.sample(self.b, n)

    def size(self):
        return len(self.b)
agent.py (core logic)
import torch
import random
import numpy as np
from model import Net

class Agent:
    def __init__(self, s, a):
        self.s = s
        self.a = a
        self.q = Net(s, a)
        self.t = Net(s, a)
        self.opt = torch.optim.Adam(self.q.parameters(), lr=1e-3)
        self.gamma = 0.99
        self.eps = 1.0

    def act(self, state):
        if random.random() < self.eps:
            return random.randrange(self.a)
        state = torch.FloatTensor(state)
        return torch.argmax(self.q(state)).item()
